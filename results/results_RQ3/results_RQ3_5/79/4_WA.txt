import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        String s = sc.next();
        boolean flag = false;
        for(int i=0;i<n;++i){
            if(s.charAt(i) == 'o') {
                flag = true;
            }else if(s.charAt(i) == 'x'){
                System.out.println("NO");
                return;
            }
        }
        if(flag) { // Check if there's at least one 'o'
            System.out.println("Yes"); // Change "Yes" to "Yes" with lowercase Y
        } else { 
            System.out.println("NO"); // Otherwise, output "NO"
        }
    }
}
