import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        long a = sc.nextLong();
        long b = sc.nextLong();
        long cnt = 0; // use long instead of int
        while (a != 0 && b != 0) { // add condition to handle when either a or b is zero
            cnt ++;
            if (a >= b) { // change '>' to '>='
                cnt += (a - 1) / b;
                a -= (a - 1) / b * b;
            } else {
                cnt += (b - 1) / a;
                b -= (b - 1) / a * a;
            }
        }
        System.out.println(cnt);
    }
}
