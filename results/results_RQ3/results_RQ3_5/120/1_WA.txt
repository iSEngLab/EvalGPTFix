import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        long A = sc.nextLong();
        long B = sc.nextLong();
        long sum = 0;
        if (A < B) {
            long tmp = A;
            A = B;
            B = tmp;
        }
        while (A != 0 && B != 0) {
            long quotient = A / B;
            sum += quotient;
            A -= quotient * B;
            if (A < B) {
                long tmp = A;
                A = B;
                B = tmp;
            }
        }
        System.out.println(sum);
    }
}
