import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        String s = sc.next();
        boolean flag = false;
        for(int i=0;i<n;++i){
            if(s.charAt(i) == 'o') {
                flag = true;
            }else if(s.charAt(i) == 'x'){
                System.out.println("NO");
                return;
            }
        }
        if (flag){
            int countO = 0;
            // fix: check if the number of 'o' is within the range [8,15]
            for (int i = 0; i < n; ++i){
                if(s.charAt(i) == 'o'){
                    countO++;
                }
            }
            if(countO >= 8 && countO <= 15 ) System.out.println("Yes");
            else if (countO + (15 - n) >= 8) System.out.println("Yes");
            else System.out.println("NO");
        }else{
            // fix: output "Yes" if the string has only o's and its length is less than 8
            if(n < 8) System.out.println("Yes");
            else System.out.println("NO");
        }
    }
}
