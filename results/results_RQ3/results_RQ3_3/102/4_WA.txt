import java.util.Scanner;

public class Main{
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String s = sc.next();

        int x = -1;
        int y = -1;

        for(int i = 0; i < s.length(); i++) {
            if(s.charAt(i) == 'B') {
                if(x == -1) {
                    x = i;
                } else {
                    y = i;
                    break;
                }
            }
        }

        if(x == -1 || y == -1 || Math.abs(x-y) % 2 == 0) {
            System.out.println("NO");
            return;
        }

        x = s.indexOf('R');
        y = s.indexOf('R', x + 1);
        int z = s.indexOf('K');

        if(x < z && z < y){
            System.out.println("YES");
            return;
        }
        System.out.println("NO");
    }
}
