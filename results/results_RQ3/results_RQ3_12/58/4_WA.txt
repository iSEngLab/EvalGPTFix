import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        String S = sc.next();

        if (N == 0) {
            System.out.println("No");
            return;
        }

        String replaced = S.replace('-', 'o');
        if (replaced.contains("-")) {
            System.out.println("No");
        } else {
            System.out.println("Yes");
        }
    }
}
