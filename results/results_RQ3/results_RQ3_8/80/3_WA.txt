import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        String s = sc.next();
        boolean flag = false;
        for(int i=0;i<n;++i){
            if(s.charAt(i) == 'o') {
                flag = true;
                break;
            } else if(s.charAt(i) != '.') {
                System.out.println("NO");
                return;
            }
        }
        System.out.println(flag ? "YES" : "NO");
    }
}
