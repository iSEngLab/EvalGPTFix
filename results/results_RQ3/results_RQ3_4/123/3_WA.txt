import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        long a = sc.nextLong();
        long b = sc.nextLong();
        long cnt = 0;
        while (a != 0 && b != 0) {
            cnt += a / b;
            a %= b;
            if (a == 0 || b == 0) break;
            cnt += b / a;
            b %= a;
        }
        if (b == 0 && a>1) cnt-=1; // fixed condition to correctly handle rounding error
        System.out.println(cnt);
    }
}
