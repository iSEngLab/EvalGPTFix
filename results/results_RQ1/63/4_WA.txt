import java.util.*;

public class Main{

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        int N = sc.nextInt();

        String s = sc.next();

        int len = s.length();

        boolean hasGood = false;
        boolean hasPoor = false;

        for(int i = 0; i < len; i++){
            char c = s.charAt(i);
            if(c == 'o')
                hasGood = true;
            else if(c == 'x')
                hasPoor = true;
        }

        if(hasGood && !hasPoor)
            System.out.println("YES");
        else
            System.out.println("NO");
    }
}
