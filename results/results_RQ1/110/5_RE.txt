package solve;

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner reader = new Scanner(System.in);
        long a = reader.nextLong();
        long b = reader.nextLong();
        long cnt = 0;
        if (a == 0 || b == 0) {
            System.out.println("0");
            return;
        }
        while (a != b) {
            if (a > b) {
                cnt += a / b;
                a %= b;
            } else {
                cnt += b / a;
                b %= a;
            }
        }
        System.out.println(cnt);
    }
}
