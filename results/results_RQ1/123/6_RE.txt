import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        long a = sc.nextInt();
        long b = sc.nextInt();
        int cnt = 0;
        
        // Check if both numbers are equal to 1
        if (a == 1 && b == 1) {
            System.out.println(0);
            return;
        }
        
        while (a != b) {
            cnt ++;
            if (a > b) {
                cnt += (a - 1) / b;
                a -= (a - 1) / b * b;
            } else {
                cnt += (b - 1) / a;
                b -= (b - 1) / a * a;
            }
        }
        System.out.println(cnt);
    }
}
