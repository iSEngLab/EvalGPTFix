import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        String s = sc.next();
        boolean flag = false;
        for(int i=0;i<n;++i){
            if(s.charAt(i) == 'o') {
                flag = true;
                break; // Add this line to exit the loop once 'o' is found
            }else if(s.charAt(i) == 'x'){
                System.out.println("NO");
                return;
            }
        }
        if(flag) System.out.println("YES");
        else System.out.println("NO");
    }
}
