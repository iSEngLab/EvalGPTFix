import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        String s = sc.next();
        boolean flag = false;
        for(int i=0;i<n;++i){
            if(s.charAt(i) == 'o') {
                flag = true;
            }else if(s.charAt(i) == 'x'){
                System.out.println("NO");
                return;
            }
        }
        if(flag) System.out.println("Yes"); // changing "YES" to "Yes" to match the expected output
        else System.out.println("NO");
    }
}
