package AtCoder;
import java.util.*;
import java.io.*;
public class Main {
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter out = new PrintWriter(System.out);
        String S = br.readLine();
        int R1 = -1;
        int R2 = -1;
        int B = -1;
        boolean test = true;
        for (int i = 0; i < S.length(); i++) {
            if (S.charAt(i)=='R'){
               if (R1==-1) R1 = i+1;
               else R2 = i+1; 
            }
            if (S.charAt(i)=='B'){
                if (B==-1) B = i+1;
                else if ((i+1+B) % 2==0) {
                    test = false;
                    break;
                }  
            }
            if (S.charAt(i)=='K' && R1!=-1 && R2!=-1 && R1<R2 && R1<i+1 && i+1<R2) test = false;
        }
        if (test) out.println("Yes");
        else out.println("No");
        out.close();
    }
}

expect: Yes
