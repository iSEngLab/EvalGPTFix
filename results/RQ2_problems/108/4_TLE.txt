import java.io.PrintWriter;
import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner sc      = new Scanner(System.in);
        PrintWriter out = new PrintWriter(System.out);

        long a = sc.nextLong();
        long b = sc.nextLong();

        int count = 0;
        while (a != b) {
            if (a > b) {
                a -= b;
            } else {
                b -= a;
            }
            count++;
        }
        out.println(count);
        out.flush();
    }
}
