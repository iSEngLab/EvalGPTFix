import java.util.*;
import java.io.*;

public class Main {
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        PrintWriter out = new PrintWriter(System.out);
        String S = br.readLine();
        int R1 = -1;
        int R2 = -1;
        int B = -1;
        boolean test = true;
        for (int i = 0; i < S.length(); i++) {
            if (S.charAt(i)=='R'){
               if (R1==-1) R1 = i+1;
               else R2 = i+1; //Fixed: added 1 to i because R2 should be the index, not the character
            }
            if (S.charAt(i)=='B'){
                if (B==-1) B = i+1;
                else if ((i+1+B) % 2==0) test = false;  
            }
            if (S.charAt(i)=='K' && R1*R2 > 0 && R1 < i+1 && i+1 < R2) test = true; //Fixed: changed "test = false" to "test = true"
        }
        if (test) out.println("Yes"); //Fixed: changed "YES" to "Yes"
        else out.println("No"); //Fixed: changed "NO" to "No"
        out.close();
    }
}
