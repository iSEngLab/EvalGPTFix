import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        long a = sc.nextLong(); //fixed to accept long input
        long b = sc.nextLong(); //fixed to accept long input
        long cnt = 0; //fixed from int to long
        while (a != b) {
            cnt ++;
            if (a > b) {
                cnt += (a - 1) / b;
                a -= (a - 1) / b * b;
            } else {
                cnt += (b - 1) / a;
                b -= (b - 1) / a * a;
            }
        }
        System.out.println(cnt);
    }
}
